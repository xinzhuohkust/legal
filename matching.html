load("统计代码和城乡划分代码.Rdata")

nations <- read_html("https://zh.wikipedia.org/zh-hans/%E4%B8%AD%E5%9C%8B%E6%B0%91%E6%97%8F%E5%88%97%E8%A1%A8") %>% 
  html_element(xpath = '//*[@id="mw-content-text"]/div[1]/table[2]') %>% 
  html_table() %>% 
  pull("中文名称") %>% 
  str_remove_all("\\[注 1\\]") # name of nations

code <- codes %>% 
  filter(年份 %in% 2010:2017) %>% 
  select(省级行政单位:区级行政单位) %>% 
  mutate(省_去除后缀 = str_remove_all(省级行政单位, "(?<=.)(维吾尔自治区|壮族自治区|回族自治区|省|市|自治区)$"),
         市_去除后缀 = str_remove_all(市级行政单位, "(?<=.)(自治州|市|盟|地区|州)$") |> str_remove_all(nations_regex) |> str_remove_all("蒙古"),
         区_去除后缀 = str_remove_all(区级行政单位, "(?<=.)(市|区|县|自治县|旗|自治旗|特区|林区|经济开发区|经济技术开发区|高新技术产业开发区|管委会|城乡一体化示范区)$") |> str_remove_all(nations_regex))

code_list <- code %>% 
  as.list() %>% 
  map(unique) 

code_list$市_去除后缀 = code_list$市_去除后缀[!code_list$市_去除后缀 == "县"]
code_list$区_去除后缀 = code_list$区_去除后缀[str_length(code_list$区_去除后缀) > 1]
code_list$区_去除后缀 = code_list$区_去除后缀[code_list$区_去除后缀 != "公安" & !code_list$区_去除后缀 %in% c(code_list$市级行政单位, code_list$省级行政单位)]

protests <- list.files() %>% 
  .[str_detect(., "protest")] %>% 
  rio::import(setclass = "tibble") %>% 
  mutate(date = map_chr(event_id, ~ str_extract(., "(?<=_).*")) |> date(), 
         year = year(date)) 

plan(multisession)
# plan(sequential)

protests <- protests %>% 
  mutate(province = map_chr(.progress = TRUE, posts, ~ str_extract(., str_c(code_list$省_去除后缀, collapse = "|"))),
         city = map(.progress = TRUE, posts, ~ str_extract_all(., str_c(code_list$市_去除后缀, collapse = "|"))),
         district = map(.progress = TRUE, posts, ~ str_extract_all(., str_c(code_list$区_去除后缀, collapse = "|"))),
         city = map_chr(.progress = TRUE, city, function(x) {unlist(x) %>% unique() %>% str_c(collapse = "_")}),
         district  = map_chr(.progress = TRUE, district, function(x) {unlist(x) %>% unique() %>% str_c(collapse = "_")}))

set.seed(230111)

sampled <- protests %>% 
  filter(city != "" | !is.na(province)) %>% 
  sample_frac(size = 100 / nrow(.))
