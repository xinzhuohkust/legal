analyzer=ChineseAnalyzer()

class legal_schema(SchemaClass):
    title=TEXT(stored=True, analyzer=analyzer)
    content=TEXT(stored=True, analyzer=analyzer)
    court=TEXT(stored=True, analyzer=analyzer)
    category=TEXT(stored=True, analyzer=analyzer)

if not os.path.exists("legal_indexing"):
    os.mkdir("legal_indexing")

schema=legal_schema()

index=create_in("legal_indexing", schema, indexname='document_index')

writer = index.writer()

for index, row in df.iterrows():
    writer.add_document(content=row['content'], title=row['title'], court=row['court'])

writer.commit()

index=open_dir("legal_indexing", indexname='document_index')

searcher = index.searcher(weighting=BM25F(B=0.75, content_B=1.0, K1=1.5))

query = QueryParser("content", index.schema).parse("") # single

query = MultifieldParser(["content", "title"], index.schema).parse("") # multi

result = searcher.search(query)

server <- function(input, output) {
    search_results <- eventReactive(input$search, {
        result <- get_result(input$query)

        if (length(result) == 0 | is.character(result)) {
            tibble("No Result Found!")
        } else {
            return(tibble(content = map_chr(0:9, \(x) result[x]$highlights("content")), score = map_dbl(0:9, \(x) result[x]$score)))
        }
    })

    # highlight_keywords <- function(text, keyword) {
    # highlighted <- gsub(paste0("\\b(", keyword, ")\\b"), "<mark>\\1</mark>", text, ignore.case = TRUE)
    # return(highlighted)
    # }

    output$results <- renderDT(
        {
            search_results()
        },
        options = list(pageLength = 5)
    )
}

shinyApp(ui, server)
